207 {
208  // initiate a new conversion
209  initialize_memory_write(0);
210  spi_write(0b10000000 | channel_number); // start a conversion
211  finish_memory_write();
212

251 {

253  initialize_memory_read(address);

263 float convert_to_signed_float(unsigned char results[4])
264 {
265  // Get the last 24 bits of the results (the first 8 bits are status bits)
266  long x = 0L;
267  x= x | ((unsigned long) results[2]<<16)
268  | ((unsigned long) results[1]<<8)
269  | ((unsigned long) results[0]);
270 
271  // Convert a 24-bit two's complement number into a 32-bit two's complement number
272  boolean sign;
273  if ((results[2]&0b10000000)==128) sign=true;
274  else sign=false;
275  if (sign) x=x | 0xFF000000;
276 
277  return float(x);
278 }

308 // Translate the fault byte into usable fault data and print it out
309 void print_fault_data(unsigned char fault_byte)
310 {
311  Serial.print(" FAULT DATA=");
312  Serial.println(fault_byte,BIN);
313 
314  if ((fault_byte&SENSOR_HARD_FAILURE)>0) Serial.println("*SENSOR HARD FALURE*");
315  if ((fault_byte&ADC_HARD_FAILURE)>0) Serial.println("*ADC_HARD_FAILURE*");
316  if ((fault_byte&CJ_HARD_FAILURE)>0) Serial.println("*CJ_HARD_FAILURE*");
317  if ((fault_byte&CJ_SOFT_FAILURE)>0) Serial.println("*CJ_SOFT_FAILURE*");
318  if ((fault_byte&SENSOR_ABOVE)>0) Serial.println("*SENSOR_ABOVE*");
319  if ((fault_byte&SENSOR_BELOW)>0) Serial.println("*SENSOR_BELOW*");
320  if ((fault_byte&ADC_RANGE_ERROR)>0) Serial.println("*ADC_RANGE_ERROR*");
321  if ((fault_byte&VALID)!=1) Serial.println("!!!!!!! INVALID READING !!!!!!!!!");
322  if (fault_byte==0b11111111) Serial.println("&&&&&&&&&& CONFIGURATION ERROR &&&&&&&&&&&&");
323  Serial.println(" ");
324 }
