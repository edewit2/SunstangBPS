64  * Function twi_init
65  * Desc readys twi pins and sets twi bitrate
66  * Input none
67  * Output none
68  */
69 void twi_init(void)
70 {
71  twi_commBuffer = 0;
72 
73  // initialize state
74  twi_state = TWI_READY;
75  twi_sendStop = true; // default value
76  twi_inRepStart = false;
77 
78  // activate internal pullups for twi.
79  digitalWrite(SDA, 1);
80  digitalWrite(SCL, 1);
81 
82  // initialize twi prescaler and bit rate
83  cbi(TWSR, TWPS0);
84  cbi(TWSR, TWPS1);
85  TWBR = ((F_CPU / TWI_FREQ) - 16) / 2;
86 
87  /* twi bit rate formula from atmega128 manual pg 204
88  SCL Frequency = CPU Clock Frequency / (16 + (2 * TWBR))
89  note: TWBR should be 10 or higher for master mode
90  It is 72 for a 16mhz Wiring board with 100kHz TWI */
91 
92  // enable twi module, acks, and twi interrupt
93  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA);
94 }

98 /*
99  * Function twi_readFrom
100  * Desc attempts to become twi bus master and read a
101  * series of bytes from a device on the bus
102  * Input address: 7bit i2c device address
103  * data: pointer to byte array
104  * length: number of bytes to read into array
105  * sendStop: Boolean indicating whether to send a stop at the end
106  * Output number of bytes read
107  */
108 uint16_t twi_readFrom(uint8_t address, uint8_t* data, uint16_t length, uint8_t sendStop)
109 {
110  uint16_t i;
111 
112 
113  // wait until twi is ready, become master receiver
114  while(TWI_READY != twi_state){
115  continue;
116  }
117  twi_state = TWI_MRX;
118  twi_sendStop = sendStop;
119  // reset error state (0xFF.. no error occured)
120  twi_error = 0xFF;
121 
122  // initialize buffer iteration vars
123  twi_commBuffer = data;
124  twi_commBufferIndex = 0;
125  twi_commBufferLength = length-1; // This is not intuitive, read on...
126  // On receive, the previously configured ACK/NACK setting is transmitted in
127  // response to the received byte before the interrupt is signalled.
128  // Therefor we must actually set NACK when the _next_ to last byte is
129  // received, causing that NACK to be sent in response to receiving the last
130  // expected byte of data.
131 
132  // build sla+w, slave device address + w bit
133  twi_slarw = TW_READ;
134  twi_slarw |= address << 1;
135 
136  if (true == twi_inRepStart) {
137  // if we're in the repeated start state, then we've already sent the start,
138  // (@@@ we hope), and the TWI statemachine is just waiting for the address byte.
139  // We need to remove ourselves from the repeated start state before we enable interrupts,
140  // since the ISR is ASYNC, and we could get confused if we hit the ISR before cleaning
141  // up. Also, don't enable the START interrupt. There may be one pending from the
142  // repeated start that we sent outselves, and that would really confuse things.
143  twi_inRepStart = false; // remember, we're dealing with an ASYNC ISR
144  do
145  {
146  TWDR = twi_slarw;
147  } while (TWCR & _BV(TWWC));
148  TWCR = _BV(TWINT) | _BV(TWEA) | _BV(TWEN) | _BV(TWIE); // enable INTs, but not START
149  }
150  else
151  // send start condition
152  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA) | _BV(TWINT) | _BV(TWSTA);
153 
154  // wait for read operation to complete
155  while(TWI_MRX == twi_state){
156  continue;
157  }
158 
159  if (twi_commBufferIndex < length)
160  length = twi_commBufferIndex;
161 
162 
163  return length;
164 }

168  * Function twi_writeTo
169  * Desc attempts to become twi bus master and write a
170  * series of bytes to a device on the bus
171  * Input address: 7bit i2c device address

175  * sendStop: boolean indicating whether or not to send a stop at the end
176  * Output 0 .. success
177  * 1 .. length to long for buffer

179  * 3 .. data send, NACK received
180  * 4 .. other twi error (lost bus arbitration, bus error, ..)

297 ISR(TWI_vect)
298 {
299  switch(TW_STATUS){
300  // All Master
301  case TW_START: // sent start condition
302  case TW_REP_START: // sent repeated start condition
303  // copy device address and r/w bit to output register and ack
304  TWDR = twi_slarw;
305  twi_reply(1);
306  break;
307 
308  // Master Transmitter
309  case TW_MT_SLA_ACK: // slave receiver acked address
310  case TW_MT_DATA_ACK: // slave receiver acked data
311  // if there is data to send, send it, otherwise stop
312  if(twi_commBufferIndex < twi_commBufferLength){
313  // copy data to output register and ack
314  TWDR = twi_commBuffer[twi_commBufferIndex++];
315  twi_reply(1);
316  }else{
317  if (twi_sendStop)
318  twi_stop();
319  else {
320  twi_inRepStart = true; // we're gonna send the START
321  // don't enable the interrupt. We'll generate the start, but we
322  // avoid handling the interrupt until we're in the next transaction,
323  // at the point where we would normally issue the start.
324  TWCR = _BV(TWINT) | _BV(TWSTA)| _BV(TWEN) ;
325  twi_state = TWI_READY;
326  }
327  }
328  break;
329  case TW_MT_SLA_NACK: // address sent, nack received
330  twi_error = TW_MT_SLA_NACK;
331  twi_stop();
332  break;
333  case TW_MT_DATA_NACK: // data sent, nack received
334  twi_error = TW_MT_DATA_NACK;
335  twi_stop();
336  break;
337  case TW_MT_ARB_LOST: // lost bus arbitration
338  twi_error = TW_MT_ARB_LOST;
339  twi_releaseBus();
340  break;
341 
342  // Master Receiver
343  case TW_MR_DATA_ACK: // data received, ack sent
344  // put byte into buffer
345  twi_commBuffer[twi_commBufferIndex++] = TWDR;
346  case TW_MR_SLA_ACK: // address sent, ack received
347  // ack if more bytes are expected, otherwise nack
348  if(twi_commBufferIndex < twi_commBufferLength){
349  twi_reply(1);
350  }else{
351  twi_reply(0);
352  }
353  break;
354  case TW_MR_DATA_NACK: // data received, nack sent
355  // put final byte into buffer
356  twi_commBuffer[twi_commBufferIndex++] = TWDR;
357  if (twi_sendStop)
358  twi_stop();
359  else {
360  twi_inRepStart = true; // we're gonna send the START
361  // don't enable the interrupt. We'll generate the start, but we
362  // avoid handling the interrupt until we're in the next transaction,
363  // at the point where we would normally issue the start.
364  TWCR = _BV(TWINT) | _BV(TWSTA)| _BV(TWEN) ;
365  twi_state = TWI_READY;
366  }
367  break;
368  case TW_MR_SLA_NACK: // address sent, nack received
369  twi_stop();
370  break;
371  // TW_MR_ARB_LOST handled by TW_MT_ARB_LOST case
372 
373 
374  // All
375  case TW_NO_INFO: // no state information
376  break;
377  case TW_BUS_ERROR: // bus error, illegal stop/start
378  twi_error = TW_BUS_ERROR;
379  twi_stop();
380  break;
381  }
382 }
