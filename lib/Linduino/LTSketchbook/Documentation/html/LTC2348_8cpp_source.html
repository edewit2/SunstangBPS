73 // Calculates the voltage from ADC output data depending on the channel configuration
74 float LTC2348_voltage_calculator(int32_t data, uint8_t channel)
75 {
76  float voltage;
77  uint8_t CW;
78  switch (channel)
79  {
80  case 0:
81  CW = CWSTRUCT.LTC2348_CHAN0_CONFIG;
82  break;
83  case 1:
84  CW = CWSTRUCT.LTC2348_CHAN1_CONFIG;
85  break;
86  case 2:
87  CW = CWSTRUCT.LTC2348_CHAN2_CONFIG;
88  break;
89  case 3:
90  CW = CWSTRUCT.LTC2348_CHAN3_CONFIG;
91  break;
92  case 4:
93  CW = CWSTRUCT.LTC2348_CHAN4_CONFIG;
94  break;
95  case 5:
96  CW = CWSTRUCT.LTC2348_CHAN5_CONFIG;
97  break;
98  case 6:
99  CW = CWSTRUCT.LTC2348_CHAN6_CONFIG;
100  break;
101  case 7:
102  CW = CWSTRUCT.LTC2348_CHAN7_CONFIG;
103  break;
104  }
105 
106  switch (CW)
107  {
108  case 0:
109  voltage = 0;
110  break; // Disable Channel
111  case 1:
112  voltage = (float)data * (1.25 * VREF / 1.000) / POW2_18;
113  break;
114  case 2:
115  {
116  voltage = (float)data * (1.25 * VREF / 1.024) / POW2_17;
117  if (voltage > 5.12)
118  voltage -= 10.24;
119  break;
120  }
121  case 3:
122  {
123  voltage = (float)data * (1.25 * VREF / 1.000) / POW2_17;
124  if (voltage > 5.12)
125  voltage -= 10.24;
126  break;
127  }
128  case 4:
129  voltage = (float)data * (2.50 * VREF / 1.024) / POW2_18;
130  break;
131  case 5:
132  voltage = (float)data * (2.50 * VREF / 1.000) / POW2_18;
133  break;
134  case 6:
135  {
136  voltage = (float)data * (2.50 * VREF / 1.024) / POW2_17;
137  if (voltage > 10.24)
138  voltage -= 20.48;
139  break;
140  }
141  case 7:
142  {
143  voltage = (float)data * (2.50 * VREF / 1.000) / POW2_17;
144  if (voltage > 10.24)
145  voltage -= 20.48;
146  break;
147  }
148  }
149  return voltage;
150 }

152 // Creates 24-bit configuration word for the 8 channels.
153 uint32_t LTC2348_create_config_word()
154 {
155  uint32_t config_word;
156  config_word = (uint32_t)CWSTRUCT.LTC2348_CHAN7_CONFIG;
157  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN6_CONFIG;
158  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN5_CONFIG;
159  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN4_CONFIG;
160  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN3_CONFIG;
161  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN2_CONFIG;
162  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN1_CONFIG;
163  config_word = (config_word << 3) | (uint32_t)CWSTRUCT.LTC2348_CHAN0_CONFIG;

168 // Transmits 24 bits (3 bytes) of configuration information and
169 // reads back 24 bytes of data (3 bytes/ 24 bits for each channel)
170 // 24 bits: 18 bit data + 3 bit config + 3 bit channel number
171 // Read back is done in a new cycle
172 void LTC2348_write(uint8_t Result[24])
173 {
174  int i, j;
175  uint32_t config_word;
176  config_word = LTC2348_create_config_word();
177 
178  output_low(QUIKEVAL_CS); //! Pull CS low
179 
180  SPI.transfer((uint8_t)(config_word >> 16)); //! Transferring byte 2
181  SPI.transfer((uint8_t)(config_word >> 8)); //! Transferring byte 1
182  SPI.transfer((uint8_t)config_word); //! Transferring byte 0
183 
184  output_high(QUIKEVAL_CS); //! Pull CS high
185  delay(100);
186  output_low(QUIKEVAL_CS);
187 
188  for (i = 0; i < 24; ++i)
189  Result[i] = SPI.transfer(0); //! Receiving 24B data by transferring null information
190 
191  output_high(QUIKEVAL_CS);
192 }
