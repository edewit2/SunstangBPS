55  * Function: nvramNode_p nvramListNew(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn);
56  *
57  * PreCondition: None
58  * Input: uint16_t dataIn = WORD of data from NVM block
59  * uint8_t pecIn = Optionally use PEC or not, same as from t_RECORD_NVM_DATA->detailedRecordHeader->UsePec;
60  * uint8_t addIn = Address of device this block belongs to, same as from t_RECORD_NVM_DATA->detailedRecordHeader->DeviceAddress
61  * uint8_t cmdIn = Command for NVM block store of device, same as from t_RECORD_NVM_DATA->detailedRecordHeader->CommandCode
62  * Output: Returns a pointer to the newly created node on the stack/heap (depending on malloc implementation)
63  * Overview: Makes a new node for the Linked List
64  * Note: Note that depending on your implementation, you will need to closely
65  * monitor your stack or heap to ensure you have sufficient space to
66  * store all NVRAM data in this dynamically allocated linked list
67  * Use of malloc is required so that the node is not destroyed when the function returns
68  *******************************************************************/
69 nvramNode_p nvramListNew(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn)
70 {
71  nvramNode_p newNode = (nvramNode_p)malloc(sizeof(nvramNode_t));
72  newNode->data_store = dataIn;
73  newNode->nvram_usePEC = pecIn;
74  newNode->nvram_pmbusAddress = addIn;
75  newNode->nvram_pmbusCommand = cmdIn;
76  newNode->next = NULL;
77  return newNode;
78 }

83  * PreCondition: Must have a typeof(nvramNode_t) list available to add to and store NVM data contents into
84  * Input: uint16_t dataIn = WORD of data from NVM block
85  * uint8_t pecIn = Optionally use PEC or not, same as from t_RECORD_NVM_DATA->detailedRecordHeader->UsePec;
86  * uint8_t addIn = Address of device this block belongs to, same as from t_RECORD_NVM_DATA->detailedRecordHeader->DeviceAddress
87  * uint8_t cmdIn = Command for NVM block store of device, same as from t_RECORD_NVM_DATA->detailedRecordHeader->CommandCode
88  * nvramNode_t* nvramList = Base address for the linked list to store to (Making this a parameter allows for multiple NVM lists in the future)
89  * Output: None
90  * Overview: Adds a new data point to the master NVRAM buffer list
91  * Note: First it navigates to the end node of the linked list, then creates a new node, and links it
92  * Note that to save processing power, you could save the last linked list address and reference that directly instead of finding it dynamically each time
93  *******************************************************************/
94 void nvramListAdd(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn, nvramNode_t *nvramList)
95 {
96  nvramNode_p newNode;
97  nvramNode_p node = nvramList;
98 
99  while (node->next != NULL)
100  {
101  node = node->next;
102  }
103 
104  newNode = nvramListNew(dataIn, pecIn, addIn, cmdIn);
105  node->next = newNode;
106 }
