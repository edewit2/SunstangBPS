108 /*
109  * Read the input current of a polyphase rail
110  *
111  * polling: poll if true
112  * return: the input current
113  */
114 float LT_PMBusRail::readIin(bool polling)
115 {
116  float current = 0.0;
117  int i = 0;
118 
119  // Add up inputs from all physical devices. This
120  // may include rail/phases that are not part of the rail.
121  while (railDef_[i].address != 0)
122  {
123  for (int j = 0; j < railDef_[i].noOfPages; j++)
124  {
125  pmbus_->setPage(railDef_[i].address, railDef_[i].pages[j]);
126  current += pmbus_->readIin(railDef_[i].address, polling);
127  }
128  i++;
129  }
130 
131  return current;
132 }

203 /*
204  * Read the status word of a polyphase rail
205  *
206  * return: status word
207  */
208 uint16_t LT_PMBusRail::readStatusWord()
209 {
210  uint16_t sw = 0.0;
211  int i = 0;
212 
213  // Combine all words. Assumes 1 = notification, so that anything
214  // with a 1 is interesting.
215  while (railDef_[i].address != 0)
216  {
217  for (int j = 0; j < railDef_[i].noOfPages; j++)
218  {
219  pmbus_->setPage(railDef_[i].address, railDef_[i].pages[j]);
220  sw |= pmbus_->readStatusWord(railDef_[i].address);
221  }
222  i++;
223  }
224 
225  return sw;
226 }
