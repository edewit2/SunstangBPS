1 /*!
2 Linear Technology DC1096B Demonstration Board.
3 LTC2642: 16-Bit DAC with SPI Interface
4 
5 @verbatim
6 
7 NOTES
8 
9  Set the terminal baud rate to 115200 and select the newline terminator.
10 
11  No external power supply is required. Two on-board reference voltages of
12  2.5V and 5V are available.
13 
14  The board features LTC2642A with direct and buffered output.
15 
16  The program displays calculated voltages which are based on the reference
17  voltage used. A precision voltmeter is needed to verify the actual
18  measured voltages against the calculated voltage displayed.
19 
20  There is also an option of generating a square wave of required voltage
21  levels and frequency. This proves the fast settling time of the corresponding
22  parts.
23 
24  Explanation of Commands:
25 
26  Setup:
27  Option 1: Enter a digital value to obtain analog voltage output.
28  Option 2: Enter the frequency and voltage levels to obtain a square wave.
29  Option 3: Vary the reference voltage used.
30 
31 USER INPUT DATA FORMAT:
32  decimal : 1024
33  hex : 0x400
34  octal : 02000 (leading 0 "zero")
35  binary : B10000000000
36  float : 1024.0

136 //! Repeats Linduino loop
137 void loop()
138 {
139  int8_t user_command; // The user input command
140  uint8_t acknowledge = 0;
141  if (Serial.available()) // Check for user input
142  {
143  user_command = read_int(); // Read the user command
144  if (user_command != 'm')
145  Serial.println(user_command); // Prints the user command to com port
146  Serial.flush();
147  switch (user_command)
148  {
149  case 1:
150  acknowledge |= voltage_output();
151  break;
152  case 2:
153  acknowledge |= square_wave_output();
154  break;
155  case 3:
156  change_VREF();
157  break;
158  default:
159  Serial.println(F("Incorrect Option"));
160  }
161  if (acknowledge)
162  Serial.println(F("***** I2C ERROR *****"));
163  Serial.print(F("\n\n\t\t\tMake sure the on-board VREF jumper is set to "));
164  Serial.print(VREF);
165  Serial.println(" V");
166  Serial.println(F("\t\t\t\tPress Enter to Continue..."));
167  read_int();
168  print_prompt();
169  }
170 }

191 //! Function to generate a square wave of desired frequency and voltage ranges
192 uint8_t square_wave_output()
193 {
194  uint16_t freq;
195  float time;
196  float voltage_high, voltage_low;
197  uint16_t code_high, code_low;
198  uint8_t receive_enter; // To receive enter key pressed
199 
200  Serial.print("\nEnter voltage_high: ");
201  while (!Serial.available());
202  voltage_high = read_float();
203  Serial.print(voltage_high);
204  Serial.println(" V");
205  receive_enter = read_int();
206 
207  Serial.print("\nEnter voltage_low: ");
208  while (!Serial.available());
209  voltage_low = read_float();
210  Serial.print(voltage_low);
211  Serial.println(" V");
212  receive_enter = read_int();
213 
214  Serial.print("\nEnter the required frequency in Hz: ");
215  freq = read_int();
216  Serial.print(freq);
217  Serial.println(" Hz");
218 
219  time = (float)1000/freq;
220  Serial.print("\nT = ");
221  Serial.print(time);
222  Serial.println(" ms");
223 
224  //! Converting data into voltage
225  code_high = LTC2642_voltage_to_code(voltage_high, VREF);
226  code_low = LTC2642_voltage_to_code(voltage_low, VREF);
227 
228  while (!Serial.available()) //! Generate square wave until a key is pressed
229  {
230  LTC2642_write(LTC2642_CS, code_high);
231  delayMicroseconds(time * 500);
232  LTC2642_write(LTC2642_CS, code_low);
233  delayMicroseconds(time * 500);
234  }
235  receive_enter = read_int();
236  return 0;
237 }
