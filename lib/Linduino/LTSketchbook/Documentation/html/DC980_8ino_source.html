395 //! soft-connect DACn to its controlled node
396 //! @return int
397 int ltc2970_soft_connect_dac(int dac_number)
398 {
399  uint16_t return_val;
400  if (dac_number == 0)
401  {
402  // check for existing faults
403  return_val = smbus->readWord(ltc2970_i2c_address,LTC2970_FAULT);
404  // if ((return_val & 0x001F) == 0x0000) {
405  if ((return_val & 0x001B) == 0x0000)
406  {
407  // make sure that the channel is not already connected
408  return_val = smbus->readWord(ltc2970_i2c_address,LTC2970_CH0_A_IDAC);

410  {
411  // the soft-connect operation can succeed with no faults, setting IDAC[9] = 1
