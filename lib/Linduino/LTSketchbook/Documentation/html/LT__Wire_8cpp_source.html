120 //
121 // Originally, 'endTransmission' was an f(void) function.
122 // It has been modified to take one parameter indicating
123 // whether or not a STOP should be performed on the bus.
124 // Calling endTransmission(false) allows a sketch to
125 // perform a repeated start.
126 //
127 // WARNING: Nothing in the library keeps track of whether
128 // the bus tenure has been properly ended with a STOP. It
129 // is very possible to leave the bus in a hung state if
130 // no call to endTransmission(true) is made. Some I2C
131 // devices will behave oddly if they do not see a STOP.
132 //
133 uint8_t TwoWire::endTransmission(uint8_t sendStop)
134 {
135  // transmit buffer (blocking)
136  int8_t ret = twi_writeTo(txAddress, commBuffer, commBufferLength, 1, sendStop);
137 
138  free(commBuffer);
139  commBuffer = 0;
140  commBufferTxMax = 0;
141  // reset tx buffer iterator vars
142  commBufferIndex = 0;
143  commBufferLength = 0;
144  // indicate that we are done transmitting
145  transmitting = 0;
146  return ret;
147 }
